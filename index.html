export default async function handler(req, res) { res.setHeader('Access-Control-Allow-Origin', '*'); res.setHeader('Access-Control-Allow-Methods', 'POST, OPTIONS'); res.setHeader('Access-Control-Allow-Headers', 'Content-Type'); if (req.method === 'OPTIONS') return res.status(200).end(); try { const { text } = req.body; const apiKey = process.env.GEMINI_API_KEY?.trim(); if (!apiKey) return res.status(200).json({ error: "APIキー未設定" }); const prompt = `あなたはタイ語と言語学の専門家です。 【絶対ルール】 1. 入力が日本語ならタイ語へ、タイ語なら日本語へ翻訳せよ。 2. "full_ipa" と "full_katakana" は、翻訳の方向に関わらず、必ず「タイ語」の発音を記述せよ。 - 入力が「元気？」なら、結果の「สบายดีไหม」の発音を書く。 - 入力が「สบายดีไหม」なら、そのままその発音を書く。 - ここに日本語の意味や「？」を決して書くな。 3. カタカナ記号：平声:→, 高声:↑, 低声:↓, 下がる:↘, 上がる:↗, 長音:〜 スキーマ: { "full_translation": "翻訳後の文全体", "full_ipa": "タイ語側の音節区切りIPA [例: sa-baai-dii-mǎi]", "full_katakana": "タイ語側の声調記号付カタカナ [例: サバーイ→ ディー→ マイ↗]", "target_lang": "翻訳後言語 (ja/th)", "words": [{"thai":"単語","reading":"カナ","ipa":"[ipa]","meaning":"意味"}] } 対象テキスト: ${text}`; const response = await fetch( `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${apiKey}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }], generationConfig: { responseMimeType: "application/json" } }) } ); const data = await response.json(); const resultText = data.candidates?.[0]?.content?.parts?.[0]?.text; let jsonContent = resultText.trim(); if (jsonContent.includes('```')) { jsonContent = jsonContent.split('```')[1].replace(/^json/, '').trim(); } return res.status(200).json(JSON.parse(jsonContent)); } catch (err) { return res.status(200).json({ error: "エラー", detail: err.message }); } }
